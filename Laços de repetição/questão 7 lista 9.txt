#include<stdio.h>

/*. Faça um algoritmo que apresente as seguintes opções ao usuário:
 T: calcular a área de um triângulo

  printf("Você deseja calcular área de um triângulo digite T, área de um quadrado digite Q, área de um retângulo digite R, caso deseje finalizar a execução digite Z: \n");
	scanf("%c", &caracter);

		scanf("%f", &altura);
		
		//processamento

		area = base * altura;
		printf("A área do seu triângulo é: %f\n", area);
			}
		else{
			if(caracter == 'Q' ||caracter =='q'){
				printf("Informe o valor correspondente ao lado do seu quadrado: ");
				scanf("%f", &lado);
				
				//processamento 
				
				area = lado * lado;
				printf("A área do quadrado é: %f\n", area);
					}
			else{
				if(caracter == 'R' || caracter == 'r'){
				printf("Informe o valor correspondente a base do retângudo: ");
				scanf("%f", &base);
				printf("Informe i valor correspondente a altura do retângulo: ");
				scanf("%f", &altura);

				//processamento
			
				area = base * altura; 
				printf("A área do retângulo é: %f\n", area);
        }
					else{
						if (caracter == 'Z' || caracter == 'z'){
						printf("Fim da execução\n");
			
						}
					}
				}
			}
  
      printf("Caso desejar executar novamente com uma outra formula digite o caracter T, Q OU R.\n");

  }while(caracter != 'Z');
	return 0;
}
