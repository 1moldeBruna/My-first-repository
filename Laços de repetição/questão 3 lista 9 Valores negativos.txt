#include<stdio.h>

/* Faça um algoritmo que leia n valores inteiros e escreva quantos desses valores são negativos.*/

int main() {

	int n=0, valores_negativos=0, valor, contador=0; 

	printf("Quantos valores serão lidos? \n");
	scanf("%d", &n);

	do{

	printf("Informe o valor para ser lido: \n");
	scanf("%d", &valor);

	if(valor<0){

	valores_negativos = valores_negativos +1;
		}
	
	contador = contador +1;
	
	}while(contador<n);

	printf("O número de valores negativos é: %d", valores_negativos);
		
	return 0;
}
