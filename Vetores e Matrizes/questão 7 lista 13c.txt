#include<stdio.h>

/*(MAT 88) Dada uma sequência de n números reais, determinar os números que compõem a
sequência e o número de vezes que cada um deles ocorre na mesma. */

//questão 7 lista 13c
int main () {

	//seçao 1
	float s;
	int n, i, j, marcacao;
	int t[n];
  	float v[n];

	printf("Informe quantos valores irão conter a sequencia de números reais: \n");
	scanf("%d", &n);

	for(i=0; i<n; i++){
	t[i]=1;
  	}


	//seçao 2
	for(i=0; i<n; i++){
		printf("Digite o valor para a sequencia: ");
		scanf("%f", &s);
		v[i]=s; // adiciona o valor digitado dentro de cada espaço do vetor
	}


	//seçao 3
  	for(i=0; i<n; i++){
    	j=1;
    	for(int g=0; g<n-i-1; g++){
      		if(v[i]==v[i+j]){
        		t[i]++;
      			}
      			j++;
    		}
  	}
  

	//seçao 4
	for(i=0; i<n; i++){
    	j=1;
    	marcacao=1; //sinalização de falso
    	for(int d=0; d<i; d++){
      		if(v[i]==v[i-j]){
        	marcacao=0; //sinalização de verdadeiro 
      		}
      		j++;
    	}
    	if(marcacao==1){
        printf("\nO valor %.1f ocorreu %d vezes", v[i], t[i]);
    		}
	}

	return 0;
}
