#include<stdio.h>

//questão2 lista -  1Algoritmo que calcula o volume da lata de óleo
int main () {

	//declaração de variáveis 
	int raio, altura;  //variáveis de entrada 
	float volume;  //variável de saída

	printf("Digite o raio da Lata de óleo: ");
	scanf("%d", &raio);

	printf("Digite a altura da lata de óleo: ");
	scanf("%d", &altura);
	
	//processamento
	volume = 3.14 * raio * raio * altura;
	printf("O volume da lata de óleo é: %f", volume);
	
	return 0;
}
