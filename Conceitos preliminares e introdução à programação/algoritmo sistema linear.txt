#include<stdio.h>
#include<math.h>

//questão 2 lista 3 algoritmo sistema linear 
int main () {

//Declaração das variáveis
	int valorA, valorB, valorC, valorD, valorE, valorF;  //variáveis de entrada
	float valorX, valorY;  //variáveis de saída

	printf("Informe o valor de a: ");
	scanf("%d", &valorA);

	printf("Informe o valor de b: ");
	scanf("%d", &valorB);
	
	printf("Informe o valor de c: ");
	scanf("%d", &valorC);

	printf("Informe o valor de d: ");
	scanf("%d", &valorD);

	printf("Informe o valor de e: ");
	scanf("%d", &valorE);

	printf("Informe o valor de f: ");
	scanf("%d", &valorF);

	//processamento 
	
	valorX = ((valorC * valorE) - (valorB * valorF)) / ((valorA * valorE) - (valorB * valorD)); 

	valorY = ((valorA * valorF) - (valorC * valorD)) / ((valorA * valorE) - (valorB * valorD)); 

	printf("X = : %f", valorX);
	
	printf("Y = : %f", valorY);

return 0;
}
