#include<stdio.h>
#include<math.h>

//questão3 lista1.algoritmo que lê os lados e calcula a área de um triângulo
int main () {

	//declaração das variáveis 
	int ladoA, ladoB, ladoC;  //variável de entrada
	double valorDeP; //variável de processamento
	double operacoesDentroDaRaiz; // variável de processamento
	float area;  //variável de saída

	printf("Informe o lado A do triângulo: ");
	scanf("%d", &ladoA);

	printf("Informe o lado B do triângulo: ");
	scanf("%d", &ladoB);

	printf("Informe o lado C do triângulo: ");
	scanf("%d", &ladoC);

	//processamento 
	valorDeP = (ladoA + ladoB + ladoC) / 2;
	
	operacoesDentroDaRaiz = valorDeP * (valorDeP - ladoA) * (valorDeP - ladoB) * (valorDeP - ladoC);

	area = sqrt(operacoesDentroDaRaiz);

	printf("a área do triângulo é: %f", area);

	return 0;
}
