 #include<stdio.h>

//Questão 1 - lista 1. Algoritmo que lê o raio e calcula a área e o comprimento de uma circunferência
int main() {
	int raio;   //variável de entrada
	float area, comprimento;  //variável de saída

	printf("Informe o raio da circunferencia: ");
	scanf ("%d", &raio);

	//processamento
	area = 3.14 * raio * raio;
    	comprimento = 2 * 3.14 * raio;

	printf("Área: %f", area );
	printf("Comprimento: %f", comprimento );

	return 0;
}
