#include<stdio.h>
#include<math.h>

//questão 4 lista 4. algoritmo que lê 2 valores e informa qual é o maior valor 
int main () {

	//declaração das variáveis 
	int valorA, valorB;   //variáveis de entrada
	float valorMaior;   //variável de saída;

	printf("Informe um valor: ");
	scanf("%f", &valorA);

	printf("Informe um valor: ");
	scanf("%f", &valorB);

	//processamento
	
	valorMaior = (valorA + valorB + abs(valorA - valorB)) / 2;
	printf("O valor maior é: %d", valorMaior);

	return 0;
}
