#include<stdio.h>
#include<math.h>

//questão 4 lista 1. algoritmo que lê os catetos de um triângulo retangulo e calcula a hipotenusa
int main () {

	//declaração das variáveis 
	int catetoA, catetoB;  //variáveis de entrada
	double catetosAoQuadrado;
	float hipotenusa;   //variável de  saída 

	printf("Informe o valor do cateto oposto ao angulo: ");
	scanf("%d", &catetoA);

	printf("Informe o valor do cateto adjacente ao angulo: ");
	scanf("%d", &catetoB);

	//processamento 
	catetosAoQuadrado = (catetoA * catetoA) + (catetoB * catetoB);

	hipotenusa = sqrt(catetosAoQuadrado);

	printf("O valor da hipotenusa é: %f", hipotenusa);

	return 0;
}
