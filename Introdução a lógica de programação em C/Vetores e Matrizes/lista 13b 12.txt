#include<stdio.h>

/*Escrever um algoritmo que lê uma matriz M(12,13) e divida todos os 13 elementos
de cada uma das 12 linhas de M pelo maior elemento em módulo daquela linha.
Escrever a matriz lida e a modificada.*/
int main () {

	//declaração das variáveis
	int a=12, b=13, valor_maior[12];
	float m[a][b];
	float dm[a][b];
	
	//obtenção dos valores da matriz
	for(int i=0; i<a; i++){
    for(int j=0; j<b; j++){

    printf("Informe o valor para a linha %d e coluna %d da matriz\n", i, j);
    scanf("%f", &m[i][j]);
    }
	}

	//mostro para o usuário a matriz lida
	printf("Os valores lidos para a matriz foram:\n");
	for(int i=0; i<a; i++){
	  for(int j=0; j<b; j++){
	    printf("%f ", m[i][j]);
		}
	  printf("\n");
	}


	//leitura de todos os valores maior de cada linha 
	for(int i=0; i<a; i++){
      	valor_maior[i] = m[i][0];
		for(int j=1; j<b; j++){
		//usei ciclo linha e coluna para passar por todos os valores
    	if(m[i][j] > valor_maior[i]){
        valor_maior[i] = m[i][j];
      }
    }     
    printf("Valor maior é: %d\n", valor_maior[i]);
	}

	//divisão dos valores do vetor pela matriz
	for(int i=0; i<a; i++){
		for(int j=0; j<b; j++){
		dm[i][j] = m[i][j]/valor_maior[i];
		}
	}

	//mostro a matriz modificada 
	printf("Matriz modificada: \n\n");
  	for(int i=0; i<a; i++){
		  for(int j=0; j<b; j++){
		    printf("%6.2f", dm[i][j]);
	    }
	  printf("\n");
	}
	
	return 0;
}
