#include<stdio.h>
/*5 lista 13b Escreva um algoritmo que leia um número inteiro A e uma matriz V 30x30 de
números. Conte quantos valores iguais a A estão na matriz. Crie, a seguir, uma
matriz X contendo todos os elementos de V diferentes de A. Mostre os resultados. */

int main () {
	
	//declaração das variáveis 
	int a, b=30, c=30, k, l;
	int matrizV[b][c], matrizX[b][c], valorA=0;

	//zerar a matriz x
	for(int k=0; k<b; k++){ 
	for(int l=0; l<c; l++){
	matrizX[k][l] = 0;
		}
	}	

	//obtenção do valor A
	printf("Informe um valor do tipo inteiro: \n");
	scanf("%d", &a);

	//obtenção dos valores para a matriz
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
	printf("Informe o valor para a linha %d e coluna %d para a matriz V: \n", i, j);
	scanf("%d", &matrizV[i][j]);
		}
	}
	

	//mostro a matriz lida 
	printf("A matrizV lida foi: \n");
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
	printf("%d ", matrizV[i][j]);
		}
  	printf("\n");
  	}

	//contar quantas vezes o valor inteiro a esteve presente na matriz
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
		if(matrizV[i][j]==a){
		valorA++;
		}
	}
}
	printf("O valor %d esteve presente dentro da matriz %d vezes. \n", a, valorA);
	

	//laços para atribuir uma matriz a outra 
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
	matrizX[i][j] = matrizV[i][j];
	}
}

	//quando o valor A estiver dentro da matriz atribuo um valor aleatório para ele
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
		if(matrizV[i][j]==a){
		  matrizX[i][j] = '\0';
		}
	}
}

	//apresentação dos resultados 
	//mostro a matriz transformada
	printf("Matriz X \n");
	for(int i=0; i<b; i++){ 
	for(int j=0; j<c; j++){
	printf("%d ", matrizX[i][j]);
		}
  	printf("\n");
  	}

	return 0;
}
