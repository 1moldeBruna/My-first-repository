#include<stdio.h>
#include<math.h>

int main () {

	char caracter;
	float base, altura, lado;
	float area;

	printf("Você deseja calcular área de um triângulo digite T, área de um quadrado digite Q, área de um retângulo digite R: ");
	scanf("%c", &caracter);

	if(caracter == 'T' ||caracter == 't'){
		printf("Informe a o valor da base do triângulo: ");
		scanf("%f", &base);
		printf("Informe a altura do triângulo: ");
		scanf("%f", &altura);
		
		//processamento

		area = base * altura;
		printf("A área do seu triângulo é: %f", area);
			}
		else{
			if(caracter == 'Q' ||caracter =='q'){
				printf("Informe o valor correspondente ao lado do seu quadrado: ");
				scanf("%f", &lado);
				
				//processamento 
				
				area = lado * lado;
				printf("A área do quadrado é: %f", area);
					}
			else{
				if(caracter == 'R' || caracter == 'r'){
				printf("Informe o valor correspondente a base do retângudo: ");
				scanf("%f", &base);
				printf("Informe i valor correspondente a altura do retângulo: ");
				scanf("%f", &altura);

				//processamento
			
				area = base * altura; 
				printf("A área do retângulo é: %f", area);

						}
					}
				}
				
	return 0;
}
